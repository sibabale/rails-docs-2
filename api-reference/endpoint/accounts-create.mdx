---
title: "Create Account"
api: "POST https://api.rails.com/api/v1/accounts"
description: "Create a new bank account for a user"
---

Creates a new bank account for an existing user. The account number is automatically generated.

## Request

<ParamField body="account_type" type="string" required>
  Type of account: `checking` or `saving`
</ParamField>

<ParamField body="user_id" type="string" required>
  UUID of the account owner
</ParamField>

<ParamField body="currency" type="string">
  Currency code (defaults to `USD`)
</ParamField>

<ParamField body="organization_id" type="string">
  Organization ID (optional)
</ParamField>

<ParamField body="environment" type="string">
  Environment: `production` or `sandbox` (defaults to `production`)
</ParamField>

## Response

<ResponseField name="id" type="string">
  Unique identifier for the account (UUID)
</ResponseField>

<ResponseField name="account_number" type="string">
  Auto-generated account number
</ResponseField>

<ResponseField name="account_type" type="string">
  Type of account: `checking` or `saving`
</ResponseField>

<ResponseField name="user_id" type="string">
  ID of the account owner
</ResponseField>

<ResponseField name="balance" type="number">
  Initial account balance (0.00)
</ResponseField>

<ResponseField name="currency" type="string">
  Currency code
</ResponseField>

<ResponseField name="status" type="string">
  Account status: `active`
</ResponseField>

<ResponseField name="organization_id" type="string">
  Organization ID (if provided)
</ResponseField>

<ResponseField name="environment" type="string">
  Environment setting
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 timestamp of account creation
</ResponseField>

<ResponseField name="updated_at" type="string">
  ISO 8601 timestamp of last update
</ResponseField>

## Example

<RequestExample>

```typescript TypeScript SDK
import { Rails } from "@rails/sdk";

const rails = new Rails({
  apiKey: process.env.RAILS_API_KEY
});

const account = await rails.accounts.create({
  accountType: "checking",
  userId: "456e7890-e89b-12d3-a456-426614174000",
  currency: "USD"
});

console.log("Account created:", account.accountNumber);
console.log("Account ID:", account.id);
```

```typescript Fetch API
const response = await fetch('https://api.rails.com/api/v1/accounts', {
  method: 'POST',
  headers: {
    'Authorization': 'Bearer your_api_key_here',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    account_type: "checking",
    user_id: "456e7890-e89b-12d3-a456-426614174000",
    currency: "USD"
  })
});

const account = await response.json();
```

</RequestExample>

## Error Codes

| Code | Description |
|------|-------------|
| `validation_error` | Invalid request parameters |
| `user_not_found` | User ID does not exist |
| `unauthorized` | Invalid or missing API key |
